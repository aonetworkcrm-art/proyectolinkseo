PARTE 2: El Script en Python para Visual Studio Code (Radar de VPH)
Para alimentar este sistema con las canciones que realmente van a dejar dinero rápido, necesitas saber qué buscar. Este script en Python hace el scraping y la matemática de forma exacta utilizando variables de coma flotante limpias para calcular el rendimiento real (VPH). No requiere instalar librerías complejas de scraping que se rompen con los cambios de YouTube; utiliza la librería oficial google-api-python-client.

```python
import os
from datetime import datetime, timezone
from googleapiclient.discovery import build

# CONFIGURACIÓN
API_KEY = "TU_YOUTUBE_API_KEY_AQUÍ"  # Reemplaza con tu clave de Google Cloud
youtube = build('youtube', 'v3', developerKey=API_KEY)

def buscar_tendencias_sonic(keyword="Mix canciones mas escuchadas", max_results=15):
    # 1. Buscar los videos más recientes del nicho
    search_request = youtube.search().list(
        q=keyword,
        part="snippet",
        type="video",
        maxResults=max_results,
        order="date"  # Forzamos a buscar lo más nuevo para capturar la ola temprana
    )
    search_response = search_request.execute()
    
    videos_data = []
    ahora = datetime.now(timezone.utc)
    
    print(f"--- Analizando tracción para: '{keyword}' ---")
    print(f"{'Título':<40} | {'Vistas':<10} | {'Horas':<7} | {'VPH Agresivo':<12}")
    print("-" * 78)

    # 2. Recopilar IDs para obtener las estadísticas exactas de cada video
    video_ids = [item['id']['videoId'] for item in search_response.get('items', [])]
    
    if not video_ids:
        print("No se encontraron videos recientes.")
        return
        
    stats_request = youtube.videos().list(
        part="statistics,snippet",
        id=",".join(video_ids)
    )
    stats_response = stats_request.execute()

    for item in stats_response.get('items', []):
        titulo = item['snippet']['title'][:38]
        vistas = int(item['statistics'].get('viewCount', 0))
        fecha_subida_str = item['snippet']['publishedAt'].replace("Z", "+00:00")
        fecha_subida = datetime.fromisoformat(fecha_subida_str)
        
        # Cálculo exacto del tiempo transcurrido en horas (Usando flotantes precisos)
        tiempo_transcurrido = ahora - fecha_subida
        horas = float(tiempo_transcurrido.total_seconds()) / 3600.0
        
        # Evitar división por cero si el video se subió hace minutos
        if horas < 0.1:
            horas = 0.1
            
        # Cálculo del Yield VPH (Views Per Hour) con coma flotante certera
        vph = float(vistas) / horas
        
        videos_data.append({
            'titulo': titulo,
            'vistas': vistas,
            'horas': horas,
            'vph': vph
        })
    
    # 3. Ordenar de manera agresiva por el mayor VPH
    videos_data.sort(key=lambda x: x['vph'], reverse=True)
    
    for v in videos_data:
        print(f"{v['titulo']:<40} | {v['vistas']:<10} | {v['horas']:<7.2f} | {v['vph']:<12.2f}")

if __name__ == "__main__":
    # Puedes cambiar el término de búsqueda según el género o artista objetivo
    buscar_tendencias_sonic("Mix canciones mas escuchadas 2026", max_results=10)

Python
import os
from datetime import datetime, timezone
from googleapiclient.discovery import build

# CONFIGURACIÓN DE API
API_KEY = "TU_YOUTUBE_API_KEY_AQUÍ"
youtube = build('youtube', 'v3', developerKey=API_KEY)

# DICCIONARIO DE NICHOS DE ALTO CPC (Estrategia de Anunciantes)
NICHOS_CPC = {
    1: {"nombre": "Fintech & Crypto (Premium)", "keywords": "trading music, bitcoin lofi, ganar dinero online", "cpc_estimado": 12.50},
    2: {"nombre": "Bienes Raíces & Lujo (High-End)", "keywords": "luxury ambient, real estate background music, chill lounge", "cpc_estimado": 8.10},
    3: {"nombre": "Software & SaaS (Corporativo)", "keywords": "coding music python, tech background beat, focus work", "cpc_estimado": 15.30},
    4: {"nombre": "Estilo de Vida & Viajes (Medio-Alto)", "keywords": "travel vlog music, beach lounge beats, vacation chill", "cpc_estimado": 4.25},
    5: {"nombre": "Desarrollo Personal & Motivación", "keywords": "gym motivation music, alpha waves focus, mindset audio", "cpc_estimado": 3.80}
}

def mostrar_menu():
    print("\n" + "="*50)
    print(" 🕵️‍♂️ SHERLOCK HOLMES SONIC ENGINE v1.0 - MENÚ CPC 🕵️‍♂️")
    print("="*50)
    for opcion, info in NICHOS_CPC.items():
        print(f"[{opcion}] {info['nombre']} | CPC Promedio Estimado: ${info['cpc_estimado']:.2f} USD")
    print("="*50)

def ejecutar_sherlock():
    mostrar_menu()
    try:
        seleccion = int(input("Elija el nicho de anunciantes para forzar el alto CPC (1-5): "))
        if seleccion not in NICHOS_CPC:
            print("Selección inválida.")
            return
    except ValueError:
        print("Por favor, introduzca un número válido.")
        return

    nicho = NICHOS_CPC[seleccion]
    keyword_busqueda = nicho["keywords"]
    cpc_target = float(nicho["cpc_estimado"]) # Uso estricto de comas flotantes

    print(f"\n[Analizando] Buscando olas de tráfico para: '{keyword_busqueda}' apuntando a CPC de ${cpc_target:.2f}...")

    # Llamada a la API buscando videos recientes del nicho
    search_request = youtube.search().list(
        q=keyword_busqueda,
        part="snippet",
        type="video",
        maxResults=10,
        order="date"
    )
    search_response = search_request.execute()
    video_ids = [item['id']['videoId'] for item in search_response.get('items', [])]

    if not video_ids:
        print("No se encontraron videos activos en este nicho hoy.")
        return

    stats_request = youtube.videos().list(
        part="statistics,snippet",
        id=",".join(video_ids)
    )
    stats_response = stats_request.execute()

    ahora = datetime.now(timezone.utc)
    resultados = []

    print("\n" + "-"*90)
    print(f"{'Título del Video Encontrado':<40} | {'Vistas':<9} | {'Horas':<6} | {'Yield VPH':<10} | {'ROI Est.':<10}")
    print("-"*90)

    for item in stats_response.get('items', []):
        titulo = item['snippet']['title'][:38]
        vistas = int(item['statistics'].get('viewCount', 0))
        fecha_str = item['snippet']['publishedAt'].replace("Z", "+00:00")
        fecha_subida = datetime.fromisoformat(fecha_str)
        
        # Cálculos precisos con floats
        horas_transcurridas = float((ahora - fecha_subida).total_seconds()) / 3600.0
        if horas_transcurridas < 0.1: 
            horas_transcurridas = 0.1
            
        # Fórmula de Rendimiento VPH Agresivo
        vph = float(vistas) / horas_transcurridas
        
        # Multiplicador de Retorno Estimado basado en el CPC seleccionado (Fórmula matemática propia)
        # Asumimos que un VPH alto combinado con un CPC alto genera un ROI exponencial
        roi_estimado = (vph * cpc_target) / 100.0 

        resultados.append({
            'titulo': titulo,
            'vistas': vistas,
            'horas': horas_transcurridas,
            'vph': vph,
            'roi': roi_estimado
        })

    # Ordenar por el ROI Estimado más agresivo para ver qué clonar primero
    resultados.sort(key=lambda x: x['roi'], reverse=True)

    for r in resultados:
        print(f"{r['titulo']:<40} | {r['vistas']:<9} | {r['horas']:<6.1f} | {r['vph']:<10.2f} | ${r['roi']:<9.2f}")

if __name__ == "__main__":
    ejecutar_sherlock()

@app.route('/api/sherlock', methods=['GET'])
def ejecutar_sherlock_web():
    if not youtube:
        return jsonify({"error": "API Key de YouTube no configurada en el servidor"}), 500

    # Capturar la industria elegida por el usuario desde la interfaz web
    industria_elegida = request.args.get('industria', 'fintech')
    
    if industria_elegida not in INDUSTRIAS_CPC:
        return jsonify({"error": "Industria no válida"}), 400

    config_industria = INDUSTRIAS_CPC[industria_elegida]
    
    # Unimos las palabras clave semánticas para la consulta de YouTube
    query_busqueda = " ".join(config_industria["keywords"][:2]) + " 2026"
    
    # Valores matemáticos precisos con comas flotantes
    cpc_promedio = float(config_industria["cpc_min"] + config_industria["cpc_max"]) / 2.0

    # 1. Ejecutar el radar de tendencias en la API de YouTube
    search_request = youtube.search().list(
        q=query_busqueda,
        part="snippet",
        type="video",
        maxResults=8,
        order="date"  # Capturar la ola Evergreen o viral reciente
    )
    search_response = search_request.execute()
    video_ids = [item['id']['videoId'] for item in search_response.get('items', [])]

    if not video_ids:
        return jsonify({"mensaje": "No se encontraron videos recientes", "resultados": []})

    stats_request = youtube.videos().list(
        part="statistics,snippet",
        id=",".join(video_ids)
    )
    stats_response = stats_request.execute()

    ahora = datetime.now(timezone.utc)
    resultados_finales = []

    for item in stats_response.get('items', []):
        vistas = int(item['statistics'].get('viewCount', 0))
        fecha_str = item['snippet']['publishedAt'].replace("Z", "+00:00")
        fecha_subida = datetime.fromisoformat(fecha_str)
        
        # Uso estricto de floats para cálculos exactos de tiempo y rendimiento
        horas = float((ahora - fecha_subida).total_seconds()) / 3600.0
        horas = max(horas, 0.1) # Evitar división por cero
        
        vph = float(vistas) / horas
        # Cálculo del ROI potencial combinando el volumen de velocidad (VPH) y el peso del CPC
        roi_potencial = (vph * cpc_promedio) / 1000.0

        resultados_finales.append({
            "titulo_original": item['snippet']['title'],
            "canal": item['snippet']['channelTitle'],
            "vistas": vistas,
            "horas_publicado": round(horas, 2),
            "vph_velocidad": round(vph, 2),
            "roi_estimado_dolares": round(roi_potencial, 2),
            "tags_sugeridos": config_industria["tags_seo"]
        })

    # Ordenar los resultados para mostrar la mina de oro primero
    resultados_finales.sort(key=lambda x: x['roi_estimado_dolares'], reverse=True)

    return jsonify({
        "industria": config_industria["nombre"],
        "cpc_rango_analizado": f"${config_industria['cpc_min']:.2f} - ${config_industria['cpc_max']:.2f}",
        "videos_detectados": resultados_finales
    })

document.getElementById('btn-escanear').addEventListener('click', async () => {
    const industria = document.getElementById('industria').value;
    const tabla = document.getElementById('tabla-resultados');
    const badgeRango = document.getElementById('rango-analizado');
    const btn = document.getElementById('btn-escanear');

    // Feedback de carga
    btn.innerText = "Buscando Olas de Data...";
    btn.disabled = true;
    tabla.innerHTML = `<tr><td colspan="3" class="py-8 text-center text-emerald-400 animate-pulse">Sherlock analizando metadatos y VPH agresivos...</td></tr>`;

    try {
        // Llamada directa al endpoint Flask/FastAPI en Vercel
        const response = await fetch(`/api/sherlock?industria=${industria}`);
        const data = await response.json();

        if (data.error) {
            tabla.innerHTML = `<tr><td colspan="3" class="py-8 text-center text-red-400">Error: ${data.error}</td></tr>`;
            return;
        }

        badgeRango.innerText = `Rango: ${data.cpc_rango_analizado}`;
        tabla.innerHTML = ''; // Limpiar

        if (data.videos_detectados.length === 0) {
            tabla.innerHTML = `<tr><td colspan="3" class="py-8 text-center text-slate-500">No se hallaron videos recientes en este nicho.</td></tr>`;
            return;
        }

        // Renderizar los activos detectados con mayor ROI matemático primero
        data.videos_detectados.forEach(video => {
            const tr = document.createElement('tr');
            tr.className = "hover:bg-slate-700/30 transition duration-150";
            tr.innerHTML = `
                <td class="py-3 pr-2 font-medium max-w-xs truncate" title="${video.titulo_original}">
                    <span class="text-white block">${video.titulo_original}</span>
                    <span class="text-xs text-slate-400 block">${video.canal} | Subido hace ${video.horas_publicado}h</span>
                </td>
                <td class="py-3 text-right text-cyan-400 font-mono">${video.vph_velocidad} vistas/h</td>
                <td class="py-3 text-right text-emerald-400 font-bold font-mono">$${video.roi_estimado_dolares}</td>
            `;
            tabla.appendChild(tr);
        });

    } catch (error) {
        tabla.innerHTML = `<tr><td colspan="3" class="py-8 text-center text-red-500">Error de conexión con la Refinería.</td></tr>`;
        console.error(error);
    } finally {
        btn.innerText = "Escanear Olas de Tráfico";
        btn.disabled = false;
    }
});

def get_youtube_service():

    if not YOUTUBE_API_KEY:

        return None

    return build("youtube", "v3", developerKey=YOUTUBE_API_KEY)





def iso_to_datetime(iso_str):

    # Expecting format like 2023-05-27T12:34:56Z

    try:

        return datetime.datetime.strptime(iso_str, "%Y-%m-%dT%H:%M:%SZ")

    except Exception:

        return None





def float_avg(a, b):

    return float(a + b) / 2.0





def search_videos(youtube, query, max_results=5):

    # Return list of video IDs for a query

    resp = youtube.search().list(

        q=query,

        part="id",

        type="video",

        maxResults=max_results

    ).execute()

    ids = []

    for item in resp.get("items", []):

        vid = item.get("id", {}).get("videoId")

        if vid:

            ids.append(vid)

    return ids





def fetch_videos_details(youtube, video_ids):

    if not video_ids:

        return []

    resp = youtube.videos().list(

        part="snippet,statistics",

        id=','.join(video_ids),

        maxResults=len(video_ids)

    ).execute()

    return resp.get("items", [])





@app.route("/api/sherlock", methods=["GET"])

def sherlock():

    industry = request.args.get("industry", type=str)

    max_per_keyword = request.args.get("max", default=5, type=int)



    if not industry:

        return jsonify({"error": "missing industry parameter"}), 400



    industry_key = industry.strip().lower()

    if industry_key not in INDUSTRIES:

        return jsonify({"error": f"unknown industry '{industry}'"}), 400



    youtube = get_youtube_service()

    if youtube is None:

        return jsonify({"error": "YouTube API key not configured in environment (YOUTUBE_API_KEY)"}), 500



    spec = INDUSTRIES[industry_key]

    keywords = spec.get("keywords", [])

    cpc_min, cpc_max = spec.get("cpc_range", (0.0, 0.0))

    cpc_estimated = float_avg(float(cpc_min), float(cpc_max))



    # Collect video ids across keywords

    collected_ids = []

    for kw in keywords:

        ids = search_videos(youtube, kw, max_results=max_per_keyword)

        for i in ids:

            if i not in collected_ids:

                collected_ids.append(i)



    details = fetch_videos_details(youtube, collected_ids)

    results = []

    now = datetime.datetime.utcnow()



    # Assumptions for ROI calculation

    CTR = 0.01  # click-through rate from views to clicks (1%)



    for v in details:

        vid = v.get("id")

        snippet = v.get("snippet", {})

        stats = v.get("statistics", {})

        title = snippet.get("title")

        published = snippet.get("publishedAt")

        published_dt = iso_to_datetime(published) if published else None

        if published_dt is None:

            hours = float(0.0)

        else:

            elapsed = now - published_dt

            hours = float(elapsed.total_seconds()) / 3600.0



        views = int(stats.get("viewCount", 0))

        views_f = float(views)

        hours_f = float(hours) if hours > 0.0 else float(1.0)



        vph_velocidad = float(views_f) / hours_f



        # ROI potential in USD estimated as: estimated_clicks * cpc_estimated

        estimated_clicks = float(views_f) * float(CTR)

        roi_potencial = float(estimated_clicks) * float(cpc_estimated)



        results.append({

            "video_id": vid,

            "titulo": title,

            "industria": industry_key,

            "vistas_originales": views,

            "vph_velocidad": float(vph_velocidad),

            "cpc_estimado": float(cpc_estimated),

            "roi_potencial": float(roi_potencial),

            "fecha_procesado": now.isoformat() + "Z"

        })



    # Order by roi_potencial desc

    results_sorted = sorted(results, key=lambda x: float(x.get("roi_potencial", 0.0)), reverse=True)



    return jsonify({

        "industry": industry_key,

        "cpc_estimated": float(cpc_estimated),

        "results": results_sorted

    })

from flask import Flask, request, jsonify, send_from_directory
from dotenv import load_dotenv
import os
import datetime
from googleapiclient.discovery import build

load_dotenv()

app = Flask(__name__, static_folder='../public', static_url_path='')

# Load API key from environment
YOUTUBE_API_KEY = os.getenv("YOUTUBE_API_KEY")

INDUSTRIES = {
    "fintech": {
        "keywords": ["fintech", "pagos digitales", "neobanco", "banca digital"],
        "cpc_range": (2.5, 6.0)
    },
    "bienes_raices": {
        "keywords": ["bienes raíces", "inmobiliaria", "comprar casa", "inversion inmobiliaria"],
        "cpc_range": (1.8, 5.0)
    },
    "saas_software": {
        "keywords": ["saas", "software como servicio", "herramienta saas", "plataforma empresarial"],
        "cpc_range": (3.0, 8.0)
    },
    "seguros": {
        "keywords": ["seguros", "seguro de vida", "seguro de auto", "cotizar seguro"],
        "cpc_range": (2.0, 7.0)
    }
}

def get_youtube_service():
    if not YOUTUBE_API_KEY or YOUTUBE_API_KEY.strip() == "" or YOUTUBE_API_KEY == "Tu_Clave_Privada_Aqui":
        return None
    try:
        return build("youtube", "v3", developerKey=YOUTUBE_API_KEY)
    except Exception:
        return None

def float_avg(a, b):
    return float(a + b) / 2.0

@app.route('/')
def serve_index():
    return send_from_directory(app.static_folder, 'index.html')

@app.route('/<path:path>')
def serve_static(path):
    return send_from_directory(app.static_folder, path)

@app.route("/api/sherlock", methods=["GET"])
def sherlock():
    industry = request.args.get("industry", type=str)
    if not industry:
        return jsonify({"error": "missing industry parameter"}), 400

    industry_key = industry.strip().lower()
    if industry_key not in INDUSTRIES:
        return jsonify({"error": f"unknown industry '{industry}'"}), 400

    spec = INDUSTRIES[industry_key]
    cpc_min, cpc_max = spec.get("cpc_range", (0.0, 0.0))
    cpc_estimated = float_avg(float(cpc_min), float(cpc_max))
    
    now = datetime.datetime.utcnow()
    youtube = get_youtube_service()

    # MOCK DATA AUTOMÁTICO: Si no hay API Key, entramos en modo simulación forense
    if youtube is None:
        results_mock = [
            {
                "video_id": "mock_1",
                "titulo": f"Tendencias Forenses en {industry_key.upper()} 2026",
                "industria": industry_key,
                "vistas_originales": 15000,
                "vph_velocidad": 625.00,
                "cpc_estimado": float(cpc_estimated),
                "roi_potencial": float(15000.0 * 0.01 * cpc_estimated),
                "fecha_procesado": now.isoformat() + "Z"
            },
            {
                "video_id": "mock_2",
                "titulo": f"Estrategia de Automatización Avanzada - Niche {industry_key}",
                "industria": industry_key,
                "vistas_originales": 8500,
                "vph_velocidad": 354.16,
                "cpc_estimado": float(cpc_estimated),
                "roi_potencial": float(8500.0 * 0.01 * cpc_estimated),
                "fecha_procesado": now.isoformat() + "Z"
            }
        ]
        # Ordenar de mayor a menor ROI potencial
        results_sorted = sorted(results_mock, key=lambda x: float(x.get("roi_potencial", 0.0)), reverse=True)
        return jsonify({
            "industry": industry_key,
            "cpc_estimated": float(cpc_estimated),
            "results": results_sorted
        })

    # LÓGICA DE PRODUCCIÓN (SI HAY API KEY CONECTADA)
    try:
        keywords = spec.get("keywords", [])
        collected_ids = []
        
        # Buscar videos por palabra clave
        for kw in keywords:
            resp = youtube.search().list(q=kw, part="id", type="video", maxResults=5).execute()
            for item in resp.get("items", []):
                vid = item.get("id", {}).get("videoId")
                if vid and vid not in collected_ids:
                    collected_ids.append(vid)

        if not collected_ids:
            return jsonify({"industry": industry_key, "cpc_estimated": float(cpc_estimated), "results": []})

        # Detalles estadísticos de los videos capturados
        v_resp = youtube.videos().list(part="snippet,statistics", id=','.join(collected_ids), maxResults=len(collected_ids)).execute()
        details = v_resp.get("items", [])
        
        results = []
        CTR = 0.01  # 1% de clics estimados

        for v in details:
            vid = v.get("id")
            snippet = v.get("snippet", {})
            stats = v.get("statistics", {})
            title = snippet.get("title")
            published = snippet.get("publishedAt")
            
            try:
                published_dt = datetime.datetime.strptime(published, "%Y-%m-%dT%H:%M:%SZ")
                elapsed = now - published_dt
                hours = float(elapsed.total_seconds()) / 3600.0
            except Exception:
                hours = float(0.0)

            views = int(stats.get("viewCount", 0))
            views_f = float(views)
            hours_f = float(hours) if hours > 0.0 else float(1.0)

            vph_velocidad = float(views_f) / hours_f

INDUSTRIES = {

def get_youtube_service():
    if not YOUTUBE_API_KEY or YOUTUBE_API_KEY.strip() == "" or YOUTUBE_API_KEY == "Tu_Clave_Privada_Aqui":
        return None
    try:
        return build("youtube", "v3", developerKey=YOUTUBE_API_KEY)
    except Exception:

def float_avg(a, b):
    return float(a + b) / 2.0

@app.route("/api/sherlock", methods=["GET"])
def sherlock():
    industry = request.args.get("industry", type=str)
    vph_min = request.args.get("vph_min", default=0.0, type=float)
    vph_max = request.args.get("vph_max", default=99999999.0, type=float)

    if not industry:
        return jsonify({"error": "missing industry parameter"}), 400

    industry_key = industry.strip().lower()
    if industry_key not in INDUSTRIES:
        return jsonify({"error": f"unknown industry '{industry}'"}), 400

    spec = INDUSTRIES[industry_key]
    cpc_min, cpc_max = spec.get("cpc_range", (0.0, 0.0))
    cpc_estimated = float_avg(float(cpc_min), float(cpc_max))
    rpm_estimated = float(spec.get("rpm", 5.00))
    
    now = datetime.datetime.utcnow()
    youtube = get_youtube_service()

    # --- MOCK DATA CON FILTRADO ANTICIPADO DE VPH ---
    if youtube is None:
        raw_mock = [
            {"video_id": "mock_1", "titulo": f"Tendencias Forenses en {industry_key.upper()} 2026", "vistas_originales": 15000, "vph_velocidad": 625.00},
            {"video_id": "mock_2", "titulo": f"Estrategia de Automatización Avanzada - Niche {industry_key}", "vistas_originales": 8500, "vph_velocidad": 354.16},
            {"video_id": "mock_3", "titulo": f"Mega Viral Direct Inflow {industry_key.upper()}", "vistas_originales": 120000, "vph_velocidad": 5000.00},
            {"video_id": "mock_4", "titulo": f"Contenido frío muerto de {industry_key}", "vistas_originales": 200, "vph_voracidad": 1.50, "vph_velocidad": 1.50}
        ]
        
        results_mock = []
        CTR = 0.01

        for item in raw_mock:
            vph = float(item["vph_velocidad"])
            if not (vph_min <= vph <= vph_max):
                continue
                
            views_f = float(item["vistas_originales"])
            estimated_clicks = views_f * CTR
            roi_cpc_potencial = estimated_clicks * cpc_estimated
            roi_rpm_nativo = (views_f / 1000.0) * rpm_estimated

            results_mock.append({
                "video_id": item["video_id"],
                "titulo": item["titulo"],
                "industria": industry_key,
                "vistas_originales": item["vistas_originales"],
                "vph_velocidad": vph,
                "cpc_estimado": cpc_estimated,
                "rpm_estimado": rpm_estimated,
                "roi_potencial": roi_cpc_potencial,
                "roi_rpm_nativo": roi_rpm_nativo,
                "fecha_procesado": now.isoformat() + "Z"
            })
            
        return jsonify({
            "industry": industry_key,
            "cpc_estimated": cpc_estimated,
            "rpm_estimated": rpm_estimated,
            "results": sorted(results_mock, key=lambda x: x["vph_velocidad"], reverse=True)
        })

    # --- LÓGICA DE PRODUCCIÓN CON API REAL ---
    try:
        keywords = spec.get("keywords", [])
        collected_ids = []
        
        for kw in keywords:
            resp = youtube.search().list(q=kw, part="id", type="video", maxResults=10).execute()
            for item in resp.get("items", []):
                vid = item.get("id", {}).get("videoId")
                if vid and vid not in collected_ids:
                    collected_ids.append(vid)

        if not collected_ids:
            return jsonify({"industry": industry_key, "cpc_estimated": cpc_estimated, "rpm_estimated": rpm_estimated, "results": []})

        v_resp = youtube.videos().list(part="snippet,statistics", id=','.join(collected_ids), maxResults=len(collected_ids)).execute()
        results = []
        CTR = 0.01

        for v in v_resp.get("items", []):
            vid = v.get("id")
            snippet = v.get("snippet", {})
            stats = v.get("statistics", {})
            
            try:
                published = snippet.get("publishedAt")
                published_dt = datetime.datetime.strptime(published, "%Y-%m-%dT%H:%M:%SZ")
                elapsed = now - published_dt
                hours = float(elapsed.total_seconds()) / 3600.0
            except Exception:
                hours = 0.0

            views = int(stats.get("viewCount", 0))
            views_f = float(views)
            hours_f = hours if hours > 0.0 else 1.0
            vph_velocidad = views_f / hours_f

            if not (vph_min <= vph_velocidad <= vph_max):
                continue

            estimated_clicks = views_f * CTR
            roi_cpc_potencial = estimated_clicks * cpc_estimated
            roi_rpm_nativo = (views_f / 1000.0) * rpm_estimated

            results.append({
                "video_id": vid,
                "titulo": snippet.get("title"),
                "industria": industry_key,
                "vistas_originales": views,
                "vph_velocidad": vph_velocidad,
                "cpc_estimado": cpc_estimated,
                "rpm_estimado": rpm_estimated,
                "roi_potencial": roi_cpc_potencial,
                "roi_rpm_nativo": roi_rpm_nativo,
                "fecha_procesado": now.isoformat() + "Z"
            })

        return jsonify({
            "industry": industry_key,
            "cpc_estimated": cpc_estimated,
            "rpm_estimated": rpm_estimated,
            "results": sorted(results, key=lambda x: x["vph_velocidad"], reverse=True)
        })

    except Exception as e:
        return jsonify({"error": str(e)}), 500
